/***********************************************************************************
Function          : 发送串口数据
      comID       : 串口（DOWNLOAD = 0, PLC = 1）
      SndBuf      : 发送缓冲区，类型为字节
      Len         : 要发送的数据长度，按字节计
      Return      : TRUE / FALSE（成功 / 失败）
************************************************************************************/
		
BOOL Send( BYTE ComID, BYTE *SndBuf, WORD Len );

实例:

BYTE byArray[10] = {0x00, ….};
Enter(PLC);	                        // 申请串口资源
Send(PLC, byArray, 10);
Receive(PLC, byArray, 10, 0, 6);
Leave(PLC);	                       // 释放资源
