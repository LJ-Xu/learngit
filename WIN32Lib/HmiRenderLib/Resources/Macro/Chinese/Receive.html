/***********************************************************************************
Function          : 接收串口数据
      comID       : 串口（DOWNLOAD = 0, PLC = 1）
      RcvBuf      : 接收缓冲区，类型为字节
      Len         : 希望接收的数据长度，按字节计
      Timeout     : 总接收超时时间（0：一直等待）。 单位：ms
      TimeOutBytes: 字节间接收超时时间（一般设为6）。
      返回值      : 实际接收到的数据长度，按字节计
************************************************************************************/
		
WORD Receive( BYTE ComID, BYTE *RcvBuf, WORD Len, WORD TimeOut, BYTE TimeOutBytes);

实例:

BYTE byArray[10] = {0x00, ….};
Enter(PLC);	                        // 申请串口资源
Send(PLC, byArray, 10);
Receive(PLC, byArray, 10, 0, 6);
Leave(PLC);	                        // 释放资源
