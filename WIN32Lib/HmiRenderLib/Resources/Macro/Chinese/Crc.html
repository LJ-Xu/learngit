/***********************************************************************************
Function    : 计算Crc值
  pHead     : 数据缓冲区起始地址
  nLen      : 缓冲区长度
Return      : Crc值
************************************************************************************/

UINT Crc(BYTE* pHead,  UINT nLen);

实例:

BYTE buffer[3] = {0x01, 0x02, 0x03};
UINT nCrc = Crc(buffer, 3);
