示例以XC为例子
/***********************************************************************************
Function      : 读写寄存器（用于位和寄存器）
      comID   : 通讯口（HMI_LOCAL_MCH = -1, DOWNLOAD = 0, PLC = 1, NET_0 ~ NET~5 = 4 ~ 9）
	  netID   : 网段号
      staID   : 站号
      objType : 寄存器地址类型
      add1    : 寄存器地址
      regs    : TYPE_BIT   = 1            占用1字节
              : TYPE_BYTE  = 2            占用1字节
              : TYPE_WORD  = 3            占用2字节
              : TYPE_DWORD = 4            占用4字节
      pRegs   : 数值缓冲区（长度应该与需读写的寄存器组大小匹配）
      Return  : TRUE / FALSE（成功 / 失败）
************************************************************************************/
			
BOOL  Writes(int comID, int netID, int staID, int objType, int add1, int regs, void* pRegs);

实例:

WORD wArray[10];
Writes(PLC, 1, XINJE_XC_REGS_D, 0, 10, wArray);
